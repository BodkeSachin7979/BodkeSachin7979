RFM Analysis in PostgreSQL

customer_segmentation.sql 
-- RFM Analysis for Customer Segmentation
WITH rfm_calc AS (
  SELECT
    customer_id,
    MAX(purchase_date) AS last_purchase,
    COUNT(*) AS frequency,
    SUM(purchase_amount) AS monetary,
    EXTRACT('day' FROM NOW() - MAX(purchase_date)) AS recency,
    NTILE(4) OVER (ORDER BY MAX(purchase_date) DESC) AS r_score,
    NTILE(4) OVER (ORDER BY COUNT(*) DESC) AS f_score,
    NTILE(4) OVER (ORDER BY SUM(purchase_amount) DESC) AS m_score
  FROM transactions
  GROUP BY customer_id
)
SELECT
  customer_id,
  recency,
  frequency,
  monetary,
  (r_score + f_score + m_score) AS rfm_total,
  CASE
    WHEN (r_score + f_score + m_score) >= 10 THEN 'Champions'
    WHEN (r_score + f_score + m_score) >= 8 THEN 'Loyal'
    WHEN (r_score + f_score + m_score) >= 6 THEN 'Returning'
    WHEN (r_score + f_score + m_score) >= 4 THEN 'Potential'
    ELSE 'At Risk'
  END AS customer_segment
FROM rfm_calc
ORDER BY rfm_total DESC;

EDA with Python

customer_analysis.py 
# Import libraries
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from scipy import stats

# Load and clean data
df = pd.read_csv('customer_shopping.csv')
print(f"Initial shape: {df.shape}")

# Handle missing values
df.drop_duplicates(inplace=True)
numeric_cols = df.select_dtypes(include=[np.number]).columns
df[numeric_cols] = df[numeric_cols].fillna(df[numeric_cols].mean())

# Feature engineering
df['purchase_month'] = pd.to_datetime(df['purchase_date']).dt.month
df['season'] = df['purchase_month'].apply(lambda x: 'Winter' if x in [12,1,2] else 
                                       ('Spring' if x in [3,4,5] else
                                        ('Summer' if x in [6,7,8] else 'Fall')))

# Customer segmentation
customer_stats = df.groupby('customer_id').agg({
    'transaction_id': 'count',
    'purchase_amount': ['sum', 'mean']
}).round(2)

# RFM scoring
customer_stats.columns = ['frequency', 'monetary', 'avg_purchase']
customer_stats['f_score'] = pd.qcut(customer_stats['frequency'], q=4, labels=[1,2,3,4])
customer_stats['m_score'] = pd.qcut(customer_stats['monetary'], q=4, labels=[1,2,3,4])

# Visualization
fig, axes = plt.subplots(2, 2, figsize=(14, 10))
sns.countplot(data=df, x='category', ax=axes[0,0])
axes[0,0].set_title('Transactions by Category')

sns.boxplot(data=df, x='category', y='purchase_amount', ax=axes[0,1])
axes[0,1].set_title('Purchase Distribution by Category')

sns.heatmap(df.corr(), annot=True, cmap='coolwarm', ax=axes[1,0])
axes[1,0].set_title('Correlation Matrix')

df.groupby('season')['purchase_amount'].mean().plot(kind='bar', ax=axes[1,1])
axes[1,1].set_title('Average Purchase by Season')

plt.tight_layout()
plt.show()
